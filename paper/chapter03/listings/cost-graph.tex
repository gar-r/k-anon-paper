func buildEmptyCostGraph(t *model.Table) *simple.WeightedUndirectedGraph {
    g := simple.NewWeightedUndirectedGraph(0, math.MaxFloat64)
    for i := range t.GetRows() {
        node := simple.Node(i)
        g.AddNode(node)
    }
    return g
}
